# Перемещение по истории в Git
# 1. Перейти в директорию репозитория
cd ~/DockerGitCourse

# 2. Просмотреть историю коммитов в сокращенном виде
git log --oneline

# 3. Переключиться на коммит с сообщением add INFO.md
git checkout e6f625c

# 4. Посмотреть содержимое рабочей директории
ls -la

# 5. Проверить текущую ветку/состояние HEAD
git branch

# 6. Вернуться на основную ветку (main)
git checkout main

# 7. Снова проверить текущее положение
git branch
