# Основные команды Git для отмены коммитов

## Git revert
# Отмена коммита через создание нового 'обратного' коммита
git revert <commit_hash>
# Пример:
git revert aa600a43cb164408e4ad87d216bc679d097f1a6c

## Git reset
# Отмена последнего коммита полностью (--hard)
git reset --hard HEAD~
# Отмена последнего коммита, изменения остаются в рабочей директории (--mixed)
git reset HEAD~
# Отмена последнего коммита, изменения остаются в индексе (--soft)
git reset --soft HEAD~1

## Просмотр
git status
git log -p
