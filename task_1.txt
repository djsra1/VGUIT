# Введение в Git

## Основные команды

# Создаем новый проект
mkdir DockerGitCourse

# Переходим в созданную директорию
cd DockerGitCourse

# Выполняем инициализацию
git init

# Создаем файлы
echo 'Hello, Hexlet!' > README.md
echo 'Haskell Curry' > PEOPLE.md

# Смотрим статус репозитория
git status

# Добавляем README.md в индекс
git add README.md

# Коммитим изменения
git commit -m 'add README.md'

# Добавляем PEOPLE.md в индекс и коммитим
git add PEOPLE.md
git commit -m 'add PEOPLE.md'

# Проверяем финальный статус
git status
