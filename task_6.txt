# Task 6: Отмена изменений в рабочей директории
## Создание неотслеживаемых файлов
mkdir one
touch two
git status
git clean -fd  # удаление всех неотслеживаемых файлов и папок
## Отмена изменений в файлах рабочей директории
echo 'new text' > INFO.md
git status
git restore INFO.md  # отмена изменений
## Отмена изменений, подготовленных к коммиту
echo 'new text' > INFO.md
git add INFO.md
git status
git restore --staged INFO.md  # снятие изменений с индекса
git restore INFO.md  # окончательная отмена изменений
